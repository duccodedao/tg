Ứng Dụng Gửi Giao Dịch TON

Lưu ý:

- Để chạy ứng dụng này, bạn cần host nó trên một máy chủ web (ví dụ: dùng Live Server của VS Code, Apache, Nginx, hoặc GitHub Pages).

- Thay thế https://your-domain.com/tonconnect-manifest.json bằng URL thực tế của file manifest và miền ứng dụng của bạn.

- File tonconnect-manifest.json phải được đặt ở thư mục gốc của domain bạn host app. Ví dụ:


{
  "url": "https://your-domain.com",
  "name": "TON DApp của tôi",
  "iconUrl": "https:/site/tonconnect-manifest.json', // !!! QUAN TRỌNG: Thay thế bằng URL manifest thực tế của bạn
                buttonRootId: 'ton-connect-button' // ID của phần tử HTML nơi nút kết nối sẽ được gắn
            });

            // Lấy tham chiếu đến các nút
            const sendBtn = document.getElementById('sendTransactionBtn');
            const sendCommentBtn = document.getElementById('sendTransactionWithCommentBtn');

            // Lắng nghe sự kiện thay đổi trạng thái kết nối ví
            tonConnectUI.onStatusChange(walletInfo => {
                if (walletInfo) {
                    // Nếu ví đã kết nối, kích hoạt các nút gửi giao dịch
                    sendBtn.disabled = false;
                    sendCommentBtn.disabled = false;
                    console.log("Ví đã kết nối:", walletInfo.account.address);
                } else {
                    // Nếu ví đã ngắt kết nối, vô hiệu hóa các nút
                    sendBtn.disabled = true;
                    sendCommentBtn.disabled = true;
                    console.log("Ví đã ngắt kết nối.");
                }
            });

            // Lắng nghe sự kiện click cho nút "Gửi 0.01 TON" (cơ bản)
            sendBtn.addEventListener('click', async () => {
                const transaction = {
                    validUntil: Math.floor(Date.now() / 1000) + 60, // Giao dịch hợp lệ trong 60 giây
                    messages: [
                        {
                            address: "UQDu8vyZSZbAYvRRQ_jW4_0EiBGibAGq72wSZjYWRmNAGhRD", // !!! Thay thế bằng địa chỉ ví người nhận
                            amount: "10000000" // 10.000.000 nanotons = 0.01 TON
                        }
                    ]
                };

                try {
                    const result = await tonConnectUI.sendTransaction(transaction);
                    alert('Giao dịch đã gửi thành công! Tx hash: ' + result.boc);
                    console.log('Kết quả giao dịch:', result);
                } catch (e) {
                    alert('Lỗi khi gửi giao dịch: ' + e.message);
                    console.error('Lỗi khi gửi giao dịch:', e);
                }
            });

            // Lắng nghe sự kiện click cho nút "Gửi 0.01 TON với Comment"
            sendCommentBtn.addEventListener('click', async () => {
                const comment = "Xin chào từ DApp TON của tôi!"; // Nội dung comment của bạn
                
                // Mã hóa chuỗi comment thành UTF-8 bytes
                const encoder = new TextEncoder();
                const encodedComment = encoder.encode(comment);

                // Tạo một Uint8Array mới chứa opcode (4 byte 0x00000000) và comment đã mã hóa
                const payloadBytes = new Uint8Array(4 + encodedComment.length);
                payloadBytes.set([0, 0, 0, 0], 0); // Opcode 0x00000000 cho text comment
                payloadBytes.set(encodedComment, 4); // Thêm comment vào sau opcode

                // Chuyển Uint8Array sang chuỗi hex để làm payload
                const payloadHex = Array.from(payloadBytes, byte => byte.toString(16).padStart(2, '0')).join('');

                const transaction = {
                    validUntil: Math.floor(Date.now() / 1000) + 60, // Giao dịch hợp lệ trong 60 giây
                    messages: [
                        {
                            address: "UQDu8vyZSZbAYvRRQ_jW4_0EiBGibAGq72wSZjYWRmNAGhRD", // !!! Thay thế bằng địa chỉ ví người nhận
                            amount: "10000000", // 10.000.000 nanotons = 0.01 TON
                            payload: payloadHex // Payload chứa comment
                        }
                    ]
                };

                try {
                    const result = await tonConnectUI.sendTransaction(transaction);
                    alert('Giao dịch kèm comment đã gửi thành công! Tx hash: ' + result.boc);
                    console.log('Kết quả giao dịch kèm comment:', result);
                } catch (e) {
                    alert('Lỗi khi gửi giao dịch kèm comment: ' + e.message);
                    console.error('Lỗi khi gửi giao dịch kèm comment:', e);
                }
            });
        };
